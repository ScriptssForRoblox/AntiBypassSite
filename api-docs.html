API Documentation

Integre nosso sistema de chaves no seu script de Roblox ou Bot de Discord.

Endpoint de Validação

URL: https://seu-dominio.com/api/external/validate

Método: POST

Body (JSON): { "key": "SUA-KEY-AQUI" }

Exemplo Roblox (Luau)


local HttpService = game:GetService("HttpService")
local URL = "https://seu-dominio.com/api/external/validate"
local MyKey = "REAL-KEY-XXXXXX"

local function Validate()
    local success, result = pcall(function()
        return HttpService:PostAsync(URL, HttpService:JSONEncode({key = MyKey}), Enum.HttpContentType.ApplicationJson)
    end)

    if success then
        local data = HttpService:JSONDecode(result)
        if data.success then
            print("Acesso Liberado: " .. data.message)
        else
            print("Erro: " .. data.message)
        end
    else
        warn("Erro ao conectar com a API")
    end
end

Validate()

Exemplo Discord Bot (Node.js)


const axios = require('axios');

async function checkKey(userKey) {
    try {
        const res = await axios.post('https://seu-dominio.com/api/external/validate', { key: userKey });
        console.log(res.data.message); // Key válida!
    } catch (err) {
        console.log(err.response.data.message); // Key inválida...
    }
}
